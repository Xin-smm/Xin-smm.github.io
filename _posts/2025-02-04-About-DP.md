### about DP

##### 0-1 knapsack problem

_Two_dimensional Array_

```c++
#include<iostream>
#include<iomanip>
using namespace std;
const int mx=2e4+10k;
int v[mx],w[mx];
int n,x;
int f[mx][mx];
int r,c,k;
void pp()
{
	for (int r=0;r<=n;r++,cout<<endl)
	for (int c=0;c<=x;c++)
	cout<<setw(3)<<f[r][c];
}
int main()
{
	cin>>x>>n;
	for (int i=1;i<=n;i++)
		cin>>w[i]>>v[i];
	for (r=1;r<=n;r++)
	{
		for (c=0;c<=x;c++)
		{
			f[r][c]=f[r-1][c];
		}
		for (c=w[r];c<=x;c++)
			f[r][c]=max(f[r][c],f[r-1][c-w[r]]+v[r]);
	}
	cout<<f[n][x];
}
```

##### Multiple Knapsack Problem

_Repeated Storage_

```c++
#include<iostream>
using namespace std;
const int mx=1e4+10;
int v[mx],w[mx],num;
int f[mx][mx];
int r,c,i,j,k,p=1;
int n,val;
int main()
{
	cin>>n>>val;
	for (i=1;i<=n;i++)
	{
		cin>>w[p]>>v[p]>>num;
		for (j=p+1;j<=p+num-1;j++)
		 w[j]=w[p],v[j]=v[p];
		p+=num;
	}
	n=p-1;
	for (r=1;r<=n;r++)
	{
		for (c=0;c<=val;c++)
			f[r][c]=f[r-1][c];
		for (c=w[r];c<=val;c++)
			f[r][c]=max(f[r][c],f[r-1][c-w[r]]+v[r]);
	}
	cout<<f[n][val];
}
```

##### Unbounded Knapsack Problem(UKP)

```c++
#include <iostream>
#define MAX(a, b) (a > b) ? a : b
using namespace std;
const int mx = 1e5 + 10;
int m, n;
int v[mx], p[mx];
int f[mx];
int i, j, k;
int main()
{
    cin >> m >> n;
    for (i = 1; i <= n; i++)
    {
        cin >> v[i] >> p[i];
    }
    for (i = 1; i <= n; i++)
    {
        for (j = v[i]; j <= m; j++)
            f[j] = max(f[j], f[j - v[i]] + p[i]);
    }
    cout << f[m];
    return 0;
}
```

##### Mixed

```c++
#include <bits/stdc++.h>
using namespace std;
const int mx = 2e4 + 10;
int f[mx];
int w[mx], v[mx], s[mx];
int n, V;
int i;
void _01knapsack(int v, int w)
{
    for (int i = V; i >= v; i--)
        f[i] = max(f[i], f[i - v] + w);
    return;
}
void _UKP(int v, int w)
{
    for (int i = v; i <= V; i++)
        f[i] = max(f[i], f[i - v] + w);
    return;
}
void _mtpKnapsack(int v, int w, int s)
{
    if (s * v > V)
    {
        cplbag(v, w);
        return;
    }
    int k = 1;
    while (k <= s)
    {
        _01knapsack(v * k, w * k);
        s -= k;
        k *= 2;
    }
    _01knapsack(v * s, w * s);
    return;
}
int main()
{
    cin >> n >> V;
    for (i = 1; i <= n; i++)
    {
        cin >> v[i] >> w[i] >> s[i];
    }
    for (i = 1; i <= n; i++)
    {
        if (s[i] == -1)
            _01knapsack(v[i], w[i]);
        if (s[i] == 0)
            _UKP(v[i], w[i]);
        if (s[i] > 0)
            _mtpKnapsack(v[i], w[i], s[i]);
    }
    cout << f[V];
    return 0;
}
```
